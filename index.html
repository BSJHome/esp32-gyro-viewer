// ====== GROUP ====== const group = new THREE.Group(); scene.add(group); // ====== LOAD TEXTURE ====== const loader = new THREE.TextureLoader(); const frontTexture = loader.load("velux_box.png"); // ====== BOX SIZE (tilpasset dit billede format) ====== const boxW = 1.6; const boxH = 2.4; // højere fordi billedet er højt const boxD = 0.35; // ====== MATERIALER ====== const frontMat = new THREE.MeshStandardMaterial({ map: frontTexture, roughness: 0.9 }); const sideMat = new THREE.MeshStandardMaterial({ color: 0x9a8a6a, roughness: 0.9 }); // rækkefølge: højre, venstre, top, bund, front, bag const materials = [ sideMat, sideMat, sideMat, sideMat, frontMat, sideMat ]; // ====== SELVE KASSEN ====== const body = new THREE.Mesh( new THREE.BoxGeometry(boxW, boxH, boxD), materials ); group.add(body); // ====== LYS ====== const light1 = new THREE.DirectionalLight(0xffffff, 1.0); light1.position.set(3, 3, 3); scene.add(light1); const light2 = new THREE.AmbientLight(0x888888); scene.add(light2); // ====== AKSER ====== const axisLength = 1.3; const zOffset = boxD / 2 + 0.05; // X (rød) const xAxis = new THREE.ArrowHelper( new THREE.Vector3(1, 0, 0), new THREE.Vector3(0, 0, zOffset), axisLength, 0xff0000 ); group.add(xAxis); // Y (grøn) const yAxis = new THREE.ArrowHelper( new THREE.Vector3(0, 1, 0), new THREE.Vector3(0, 0, zOffset), axisLength, 0x00ff00 ); group.add(yAxis); // Z (blå) const zAxis = new THREE.ArrowHelper( new THREE.Vector3(0, 0, 1), new THREE.Vector3(0, 0, zOffset), 0.8, 0x0066ff ); group.add(zAxis); // ====== LABELS ====== function makeLabel(text, color) { const canvas = document.createElement("canvas"); canvas.width = 256; canvas.height = 128; const ctx = canvas.getContext("2d"); ctx.fillStyle = "#" + color.toString(16).padStart(6, "0"); ctx.font = "bold 80px Arial"; ctx.textAlign = "center"; ctx.textBaseline = "middle"; ctx.fillText(text, 128, 64); const texture = new THREE.CanvasTexture(canvas); const material = new THREE.SpriteMaterial({ map: texture, transparent: true, depthTest: false }); const sprite = new THREE.Sprite(material); sprite.scale.set(0.4, 0.2, 1); return sprite; } const labelX = makeLabel("X", 0xff0000); labelX.position.set(axisLength + 0.25, 0, zOffset); group.add(labelX); const labelY = makeLabel("Y", 0x00ff00); labelY.position.set(0, axisLength + 0.25, zOffset); group.add(labelY); const labelZ = makeLabel("Z", 0x0066ff); labelZ.position.set(0, 0, zOffset + 1.0); group.add(labelZ); // ====== CENTER DOT ====== const center = new THREE.Mesh( new THREE.SphereGeometry(0.05, 16, 16), new THREE.MeshBasicMaterial({ color: 0xffffff }) ); center.position.set(0, 0, zOffset); group.add(center);
